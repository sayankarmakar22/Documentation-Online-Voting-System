Election APIs Base: /election

POST /election/create

Request Body

{
  "type": "NATIONAL_LEVEL | STATE_LEVEL | LOCAL_LEVEL",
  "start_date": "DATE(YYYY-DD-MM)",
  "end_date": "DATE(YYYY-DD-MM)"
}

Response

{
  "election_id": "string"
}

Request Body

{
  "elec_id": "string",
  "status": "NOT_STARTED | IN_PROGRESS | DATE_PUBLISHED | ON_GOING | CLOSED | RESULT_PUBLISHED"
}

}
